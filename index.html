  const homePage = document.getElementById("home");
  const kavithaiPage = document.getElementById("kavithai-page");
  const aboutPage = document.getElementById("about-page");

  const list = document.getElementById("list");
  const kTitle = document.getElementById("k-title");
  const kContent = document.getElementById("k-content");

  const backBtn = document.getElementById("back");
  const aboutBtn = document.getElementById("about-btn");
  const aboutBack = document.getElementById("about-back");

  // Load kavithai list
  kavithaigal.forEach(k => {
    const div = document.createElement("div");
    div.className = "kavithai-item";
    div.textContent = k.title;

    div.onclick = () => {
      homePage.classList.add("hidden");
      aboutPage.classList.add("hidden");
      kavithaiPage.classList.remove("hidden");

      kTitle.textContent = k.title;
      kContent.textContent = k.content;
    };

    list.appendChild(div);
  });

  // Back buttons
  backBtn.onclick = () => {
    kavithaiPage.classList.add("hidden");
    homePage.classList.remove("hidden");
  };

  aboutBtn.onclick = () => {
    homePage.classList.add("hidden");
    kavithaiPage.classList.add("hidden");
    aboutPage.classList.remove("hidden");
  };

  aboutBack.onclick = () => {
    aboutPage.classList.add("hidden");
    homePage.classList.remove("hidden");
  };
